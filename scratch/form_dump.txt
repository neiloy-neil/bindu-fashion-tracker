export default function NewEntryPage() {
  const router = useRouter()
  const [branches, setBranches] = useState<Branch[]>([])
  const [categories, setCategories] = useState<Category[]>([])
  const [loading, setLoading] = useState(false)
  const [uploadingAttachment, setUploadingAttachment] = useState<Record<string, boolean>>({})
  
  const [accounts, setAccounts] = useState<any[]>([])
  const [parties, setParties] = useState<any[]>([])
  const [expenseCategories, setExpenseCategories] = useState<any[]>([])
  const [employees, setEmployees] = useState<any[]>([])
  
  const [userRole, setUserRole] = useState<string>('')
  const [missingPreviousDay, setMissingPreviousDay] = useState(false)
  const [showChecklistModal, setShowChecklistModal] = useState(false)

  const [showIncome, setShowIncome] = useState(true)
  const [showExpense, setShowExpense] = useState(true)

  const today = new Date().toISOString().split('T')[0]

  const form = useForm<FormValues>({
    resolver: zodResolver(newEntryFormSchema),
    defaultValues: {
      formMeta: { date: today, branchId: '', openingTime: '09:00', closingTime: '21:00' },
      incomeItems: [],
      transfers: [],
      payments: [],
      expenseEntries: [],
      advanceSalaries: [],
      globalNotes: '',
      actualPhysicalCash: '',
      cashDifferenceNote: '',
      eodChecklist: {
        safeLocked: false, acOff: false, shopClean: false, shuttersDown: false, cashVerified: false, signature: ''
      }
    }
  })

  const { control, handleSubmit, watch, setValue, formState: { errors } } = form

  const { fields: incomeFields, append: appendIncome, remove: removeIncome } = useFieldArray({ control, name: 'incomeItems' })
  const { fields: transferFields, append: appendTransfer, remove: removeTransfer } = useFieldArray({ control, name: 'transfers' })
  const { fields: paymentFields, append: appendPayment, remove: removePayment } = useFieldArray({ control, name: 'payments' })
  const { fields: expenseFields, append: appendExpense, remove: removeExpense } = useFieldArray({ control, name: 'expenseEntries' })
  const { fields: advanceFields, append: appendAdvance, remove: removeAdvance } = useFieldArray({ control, name: 'advanceSalaries' })

  const watchAll = watch()

  // Autosave logic
  useEffect(() => {
    const timeoutId = setTimeout(() => {
      localStorage.setItem('newEntryDraftRHF', JSON.stringify(watchAll))
    }, 1000)
  return () => clearTimeout(timeoutId)
  }, [watchAll])

  useEffect(() => {
    fetch('/api/branches').then(r => r.json()).then(data => {
      setBranches(data)
      if (data.length === 1 && !watchAll.formMeta.branchId) {
        setValue('formMeta.branchId', String(data[0].id))
      }
    })

    fetch('/api/categories?active=true').then(r => r.json()).then(data => {
      setCategories(data)
      const draft = localStorage.getItem('newEntryDraftRHF')
      if (draft) {
        try {
          const parsed = JSON.parse(draft)
          // Restore files objects properly? File objects cannot be serialized to localStorage.
          // For now, we clear files in draft.
          if (parsed.incomeItems) {
            parsed.incomeItems.forEach((i: any) => { if (i.detail) i.detail.files = [] })
          }
          form.reset(parsed)
        } catch {}
      }
    })

    Promise.all([
      fetch('/api/accounts').then(r => r.json()),
      fetch('/api/parties').then(r => r.json()),
      fetch('/api/expense-categories').then(r => r.json()),
      fetch('/api/employees').then(r => r.json())
    ]).then(([accData, partyData, expCatData, empData]) => {
      setAccounts(accData.accounts || accData)
      setParties(partyData.parties || partyData)
      setExpenseCategories(expCatData.categories || expCatData)
      setEmployees(empData.employees || empData)
    })
  }, [])

  useEffect(() => {
    if (watchAll.formMeta.branchId && watchAll.formMeta.date && categories.length > 0) {
      fetch(`/api/entries/last-balance?branchId=${watchAll.formMeta.branchId}&date=${watchAll.formMeta.date}`)
        .then(r => r.json())
        .then(data => {
          if (data.userRole) setUserRole(data.userRole)
          
          if (data.lastEntry === null) {
            setMissingPreviousDay(true)
          } else {
            setMissingPreviousDay(false)
            const openingCat = categories.find(c => c.name === 'Opening Balance')
            if (openingCat && data.userRole !== 'ADMIN') {
              const currentIncome = form.getValues('incomeItems')
              const idx = currentIncome.findIndex(i => i.categoryId === openingCat.id)
              if (idx !== -1) {
                setValue(`incomeItems.${idx}.amount`, data.actualPhysicalCash)
              } else {
                setValue('incomeItems', [
                  { id: generateId(), categoryId: openingCat.id, amount: data.actualPhysicalCash, detail: { note: 'Auto-filled by system', files: [] } },
                  ...currentIncome
                ])
              }
            }
          }
        })
    }
  }, [watchAll.formMeta.branchId, watchAll.formMeta.date, categories])

  const fakeEntry = {
    items: watchAll.incomeItems.map(i => ({ amount: Number(i.amount), category: categories.find(c => c.id === Number(i.categoryId)) || null })),
    transfers: watchAll.transfers.map(t => ({ amount: Number(t.amount) })),
    payments: watchAll.payments.map(p => ({ amount: Number(p.amount), method: p.method })),
    expenseEntries: watchAll.expenseEntries.map(e => ({ amount: Number(e.amount) })),
    advanceSalaries: watchAll.advanceSalaries.map(a => ({ amount: Number(a.amount), type: a.type }))
  }
  const totals = computeTotals(fakeEntry)

  const uploadFiles = async (files: File[]) => {
    const urls: string[] = []
    for (const file of files) {
      const formData = new FormData()
      formData.append('file', file)
      const res = await fetch('/api/upload', { method: 'POST', body: formData })
      if (!res.ok) throw new Error('Upload failed')
      const data = await res.json()
      urls.push(data.url)
    }
    return urls
  }

  const onSubmit: any = async (data: any) => {
    if (!data.eodChecklist.cashVerified) {
      setShowChecklistModal(true)
      return
    }
    
    if (missingPreviousDay && userRole !== 'ADMIN') {
      toast.error('Cannot submit entry because previous day entry is missing. Contact Admin.')
      return
    }

    setLoading(true)
    const uploadToast = toast.loading('Uploading files...')
    
    try {
      // Process items and upload files
      const itemsToSubmit = []
      for (const item of data.incomeItems) {
        if (!item.categoryId) continue
        const urls = item.detail.files.length > 0 ? await uploadFiles(item.detail.files) : []
        const note = item.detail.note.trim()
        if (note && urls.length === 0) {
          toast.error("Receipt image is required when a note is provided for Income/Expense.")
          toast.dismiss(uploadToast)
          setLoading(false)
          return
        }
        itemsToSubmit.push({
          categoryId: Number(item.categoryId),
          amount: Number(item.amount),
          receiptUrls: urls,
          notes: note || undefined
        })
      }
      
      toast.loading('Submitting entry...', { id: uploadToast })

      const payload = {
        date: new Date(data.formMeta.date),
        branchId: Number(data.formMeta.branchId),
        notes: data.globalNotes,
        actualPhysicalCash: Number(data.actualPhysicalCash),
        cashDifferenceNote: data.cashDifferenceNote,
        eodChecklist: data.eodChecklist,
        items: itemsToSubmit,
        transfers: data.transfers.map(t => ({ ...t, accountId: Number(t.accountId), amount: Number(t.amount) })),
        payments: data.payments.map(p => ({ ...p, partyId: Number(p.partyId), amount: Number(p.amount) })),
        expenseEntries: data.expenseEntries.map(e => ({ ...e, categoryId: Number(e.categoryId), amount: Number(e.amount) })),
        advanceSalaries: data.advanceSalaries.map(a => ({ ...a, employeeId: Number(a.employeeId), amount: Number(a.amount) }))
      }

      const res = await fetch('/api/entries', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      })

      if (!res.ok) {
        const errorData = await res.json()
        throw new Error(errorData.error || 'Failed to submit')
      }
      
      localStorage.removeItem('newEntryDraftRHF')
      
      // Auto-download Daily Report PDF
      try {
        const dateStr = payload.date.toISOString().split('T')[0]
        const branchName = branches.find(b => b.id === payload.branchId)?.name || 'Branch'
        const reportRes = await fetch(`/api/reports/daily?branchId=${payload.branchId}&date=${dateStr}`)
        if (!reportRes.ok) throw new Error('Failed to fetch report')
        const reportData = await reportRes.json()
        
        const { exportReportAsPdf } = await import('@/lib/exportPdf')
        await exportReportAsPdf(reportData, branchName, dateStr)
        
        toast.success('Register closed — daily report downloaded.', { id: uploadToast })
      } catch (reportErr) {
        toast.error('Register closed and saved successfully. Report download failed — you can re-download it from the Daily Report page.', { id: uploadToast, duration: 8000 })
      }

      router.push('/entries')

    } catch (err: any) {
      toast.error(err.message || 'Error saving entry', { id: uploadToast })
      setLoading(false)
    }
  }

  // Common UI classes
  const inputClass = "form-input h-10 w-full text-sm"
  const selectClass = "form-input form-select h-10 w-full text-sm"

  return (
    <div className="p-4 sm:p-8 max-w-5xl mx-auto pb-32">
      <div className="mb-6">
        <h1 className="text-2xl font-bold flex items-center gap-2">
          📝 Daily Entry
        </h1>
        <p className="text-[var(--text-secondary)] mt-1 text-sm">
          Submit the end-of-day financial report for your branch.
        </p>
      </div>

      {missingPreviousDay && userRole !== 'ADMIN' && (
        <div className="mb-6 p-4 bg-[#ef4444]/10 border border-[#ef4444]/30 rounded-lg flex items-start gap-3">
          <Lock className="text-[#ef4444] mt-0.5 flex-shrink-0" size={18} />
          <div>
            <h3 className="text-[#ef4444] font-semibold text-sm">Submission Locked</h3>
            <p className="text-[#ef4444]/80 text-xs mt-1">
              You cannot submit today's entry because the entry for the previous working day is missing. 
              Please contact an Administrator to resolve this discrepancy.
            </p>
          </div>
        </div>
      )}

      <form onSubmit={handleSubmit(onSubmit)} className="space-y-6">
        
        {/* Branch & Meta Section */}
        <div className="bg-[#131b2c] p-4 sm:p-5 rounded-lg border border-[#1e2d45]">
          <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-4">
            <div>
              <label className="form-label text-xs">Date</label>
              <input type="date" className={inputClass} {...form.register('formMeta.date')} />
              <ErrorMsg error={errors.formMeta?.date} />
            </div>
            <div>
              <label className="form-label text-xs">Branch</label>
              <select className={selectClass} {...form.register('formMeta.branchId')}>
                <option value="">Select Branch</option>
                {branches.map(b => <option key={b.id} value={b.id}>{b.name}</option>)}
              </select>
              <ErrorMsg error={errors.formMeta?.branchId} />
            </div>
            <div>
              <label className="form-label text-xs">Opening Time</label>
              <input type="time" className={inputClass} {...form.register('formMeta.openingTime')} />
            </div>
            <div>
              <label className="form-label text-xs">Closing Time</label>
              <input type="time" className={inputClass} {...form.register('formMeta.closingTime')} />
            </div>
          </div>
        </div>

        {/* Income Section */}
        <div className="bg-[#131b2c] rounded-lg border border-[#1e2d45] overflow-hidden">
          <div 
            className="flex justify-between items-center p-4 bg-[#1e2d45]/30 cursor-pointer"
            onClick={() => setShowIncome(!showIncome)}
          >
            <h2 className="text-lg font-semibold flex items-center gap-2">
              💰 Income & Collections
              <span className="text-xs bg-[#10b981]/20 text-[#10b981] px-2 py-0.5 rounded ml-2 font-mono">
                {formatCurrency(totals.totalAmount)}
              </span>
            </h2>
            {showIncome ? <ChevronDown size={20} className="text-[#8899aa]" /> : <ChevronRight size={20} className="text-[#8899aa]" />}
          </div>
          
          {showIncome && (
            <div className="p-4 sm:p-5">
              {incomeFields.map((field, idx) => {
                const catId = watchAll.incomeItems[idx]?.categoryId
                const isOpeningBal = catId && categories.find(c => c.id === Number(catId))?.name === 'Opening Balance'
                const readOnly = isOpeningBal && userRole !== 'ADMIN'
                
  return (
                  <div key={field.id} className="bg-[#0a0f18] border border-[#1e2d45] rounded-lg p-4 mb-3 relative group">
                    <button 
                      type="button" 
                      onClick={() => removeIncome(idx)}
                      className="absolute -right-2 -top-2 bg-[#ef4444] text-white p-1 rounded-full opacity-0 group-hover:opacity-100 transition-opacity hover:bg-[#dc2626] shadow-lg"
                    >
                      <X size={14} />
                    </button>
                    <div className="flex flex-col sm:flex-row gap-4">
                      <div className="flex-1">
                        <label className="form-label text-xs">Category</label>
                        <select className={selectClass} {...form.register(`incomeItems.${idx}.categoryId` as const)}>
                          <option value="">Select Category</option>
                          {categories.filter(c => c.type === 'INCOME').map(c => <option key={c.id} value={c.id}>{c.name}</option>)}
                        </select>
                        <ErrorMsg error={errors.incomeItems?.[idx]?.categoryId} />
                      </div>
                      <div className="flex-1">
                        <label className="form-label text-xs">Amount</label>
                        <input 
                          type="number" 
                          className={inputClass} 
                          placeholder="0" 
                          readOnly={!!readOnly}
                          {...form.register(`incomeItems.${idx}.amount` as const)} 
                        />
                        <ErrorMsg error={errors.incomeItems?.[idx]?.amount} />
                      </div>
                    </div>

                    <div className="mt-3 p-3 rounded-md text-xs transition-all bg-[#162033] border border-[#1e2d45]">
                      <div className="flex flex-col gap-2">
                        <div>
                          <label className="text-[#8899aa] mb-1 block">Reason / Note</label>
                          <textarea
                            className="w-full bg-[#0a0f1e] border border-[#1e2d45] rounded p-2 text-[#f0f4ff] focus:outline-none focus:border-[#00d2ff]"
                            rows={2}
                            placeholder="Enter reason..."
                            {...form.register(`incomeItems.${idx}.detail.note` as const)}
                          />
                        </div>
                        <div>
                          <label className="text-[#8899aa] mb-1 block">Receipt Images</label>
                          <input
                            type="file"
                            multiple
                            accept="image/*"
                            onChange={(e) => {
                              const files = Array.from(e.target.files || []);
                              const current = form.getValues(`incomeItems.${idx}.detail.files` as const) || [];
                              form.setValue(`incomeItems.${idx}.detail.files` as const, [...current, ...files]);
                            }}
                            className="block w-full text-xs text-[#8899aa] file:mr-4 file:py-1 file:px-3 file:rounded file:border-0 file:text-xs file:font-semibold file:bg-[#10b981] file:text-white hover:file:bg-[#059669] cursor-pointer"
                          />
                          {watchAll.incomeItems[idx]?.detail?.files?.length > 0 && (
                            <div className="flex flex-col gap-1 mt-2">
                              {watchAll.incomeItems[idx].detail.files.map((file: any, fIdx: number) => (
                                <div key={fIdx} className="flex justify-between items-center bg-[#0a0f1e] p-1.5 rounded border border-[#1e2d45]">
                                  <span className="truncate max-w-[150px] text-[#34d399]">{file.name}</span>
                                  <button type="button" onClick={() => {
                                    const current = form.getValues(`incomeItems.${idx}.detail.files` as const);
                                    current.splice(fIdx, 1);
                                    form.setValue(`incomeItems.${idx}.detail.files` as const, current);
                                  }} className="text-[#f87171] hover:text-[#dc2626]">
                                    <X size={14} />
                                  </button>
                                </div>
                              ))}
                            </div>
                          )}
                        </div>
                      </div>
                    </div>
                  </div>
                )
              })}
              <button
                type="button"
                onClick={() => appendIncome({ id: generateId(), categoryId: '', amount: '', detail: { note: '', files: [] } })}
                className="btn border border-[#1e2d45] bg-[#0a0f18] text-[#8899aa] hover:text-white hover:border-[#00d2ff]/50 w-full py-3 border-dashed"
              >
                <Plus size={16} /> Add Income Item
              </button>
            </div>
          )}
        </div>

        {/* Unified Expenses Section */}
        <div className="bg-[#131b2c] rounded-lg border border-[#1e2d45] overflow-hidden">
          <div 
            className="flex justify-between items-center p-4 bg-[#1e2d45]/30 cursor-pointer"
            onClick={() => setShowExpense(!showExpense)}
          >
            <h2 className="text-lg font-semibold flex items-center gap-2">
              💸 Expenses & Dispersals
              <span className="text-xs bg-[#ef4444]/20 text-[#ef4444] px-2 py-0.5 rounded ml-2 font-mono">
                {formatCurrency(totals.totalExpense)}
              </span>
            </h2>
            {showExpense ? <ChevronDown size={20} className="text-[#8899aa]" /> : <ChevronRight size={20} className="text-[#8899aa]" />}
          </div>
          
          {showExpense && (
            <div className="p-4 sm:p-5 space-y-8">
              
              {/* Daily Expenses */}
              <div>
                <h3 className="text-sm font-semibold text-[#8899aa] uppercase tracking-wider mb-3">Daily Expenses</h3>
                {expenseFields.map((field, idx) => (
                  <div key={field.id} className="bg-[#0a0f18] border border-[#1e2d45] rounded-lg p-4 mb-3 relative group">
                    <button type="button" onClick={() => removeExpense(idx)} className="absolute -right-2 -top-2 bg-[#ef4444] text-white p-1 rounded-full opacity-0 group-hover:opacity-100">
                      <X size={14} />
                    </button>
                    <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
                      <div>
                        <select className={selectClass} {...form.register(`expenseEntries.${idx}.categoryId` as const)}>
                          <option value="">Category</option>
                          {expenseCategories.map(c => <option key={c.id} value={c.id}>{c.name}</option>)}
                        </select>
                        <ErrorMsg error={errors.expenseEntries?.[idx]?.categoryId} />
                      </div>
                      <div>
                        <input type="number" className={inputClass} placeholder="Amount" {...form.register(`expenseEntries.${idx}.amount` as const)} />
                        <ErrorMsg error={errors.expenseEntries?.[idx]?.amount} />
                      </div>
                      <div>
                        <input type="text" className={inputClass} placeholder="Note" {...form.register(`expenseEntries.${idx}.note` as const)} />
                      </div>
                    </div>
                  </div>
                ))}
                <button type="button" onClick={() => appendExpense({ id: generateId(), categoryId: '', amount: '', note: '' })} className="text-xs text-[#00d2ff] hover:underline flex items-center gap-1">
                  <Plus size={14} /> Add Expense
                </button>
              </div>

              {/* Transfers */}
              <div>
                <h3 className="text-sm font-semibold text-[#8899aa] uppercase tracking-wider mb-3">Bank Transfers</h3>
                {transferFields.map((field, idx) => (
                  <div key={field.id} className="bg-[#0a0f18] border border-[#1e2d45] rounded-lg p-4 mb-3 relative group">
                    <button type="button" onClick={() => removeTransfer(idx)} className="absolute -right-2 -top-2 bg-[#ef4444] text-white p-1 rounded-full opacity-0 group-hover:opacity-100">
                      <X size={14} />
                    </button>
                    <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
                      <div>
                        <select className={selectClass} {...form.register(`transfers.${idx}.accountId` as const)}>
                          <option value="">To Account</option>
                          {accounts.map(a => <option key={a.id} value={a.id}>{a.name}</option>)}
                        </select>
                        <ErrorMsg error={errors.transfers?.[idx]?.accountId} />
                      </div>
                      <div>
                        <input type="number" className={inputClass} placeholder="Amount" {...form.register(`transfers.${idx}.amount` as const)} />
                        <ErrorMsg error={errors.transfers?.[idx]?.amount} />
                      </div>
                      <div>
                        <input type="text" className={inputClass} placeholder="Note/Slip No" {...form.register(`transfers.${idx}.note` as const)} />
                      </div>
                    </div>
                  </div>
                ))}
                <button type="button" onClick={() => appendTransfer({ id: generateId(), accountId: '', amount: '', note: '' })} className="text-xs text-[#00d2ff] hover:underline flex items-center gap-1">
                  <Plus size={14} /> Add Transfer
                </button>
              </div>

              {/* Advance Salaries */}
              <div>
                <h3 className="text-sm font-semibold text-[#8899aa] uppercase tracking-wider mb-3">Advance Salaries</h3>
                {advanceFields.map((field, idx) => (
                  <div key={field.id} className="bg-[#0a0f18] border border-[#1e2d45] rounded-lg p-4 mb-3 relative group">
                    <button type="button" onClick={() => removeAdvance(idx)} className="absolute -right-2 -top-2 bg-[#ef4444] text-white p-1 rounded-full opacity-0 group-hover:opacity-100">
                      <X size={14} />
                    </button>
                    <div className="grid grid-cols-1 sm:grid-cols-4 gap-4">
                      <div>
                        <select className={selectClass} {...form.register(`advanceSalaries.${idx}.employeeId` as const)}>
                          <option value="">Select Employee</option>
                          {employees.map(e => <option key={e.id} value={e.id}>{e.name}</option>)}
                        </select>
                        <ErrorMsg error={errors.advanceSalaries?.[idx]?.employeeId} />
                      </div>
                      <div>
                        <select className={selectClass} {...form.register(`advanceSalaries.${idx}.type` as const)}>
                          <option value="CASH">Cash Advance</option>
                          <option value="PRODUCT">Product Purchase</option>
                        </select>
                      </div>
                      <div>
                        <input type="number" className={inputClass} placeholder="Amount" {...form.register(`advanceSalaries.${idx}.amount` as const)} />
                        <ErrorMsg error={errors.advanceSalaries?.[idx]?.amount} />
                      </div>
                      <div>
                        <input type="text" className={inputClass} placeholder="Note" {...form.register(`advanceSalaries.${idx}.note` as const)} />
                      </div>
                    </div>
                  </div>
                ))}
                <button type="button" onClick={() => appendAdvance({ id: generateId(), employeeId: '', type: 'CASH', amount: '', productDescription: '', note: '' })} className="text-xs text-[#00d2ff] hover:underline flex items-center gap-1">
                  <Plus size={14} /> Add Advance
                </button>
              </div>

              {/* Payments */}
              <div>
                <h3 className="text-sm font-semibold text-[#8899aa] uppercase tracking-wider mb-3">Party Payments</h3>
                {paymentFields.map((field, idx) => (
                  <div key={field.id} className="bg-[#0a0f18] border border-[#1e2d45] rounded-lg p-4 mb-3 relative group">
                    <button type="button" onClick={() => removePayment(idx)} className="absolute -right-2 -top-2 bg-[#ef4444] text-white p-1 rounded-full opacity-0 group-hover:opacity-100">
                      <X size={14} />
                    </button>
                    <div className="grid grid-cols-1 sm:grid-cols-4 gap-4">
                      <div>
                        <select className={selectClass} {...form.register(`payments.${idx}.partyId` as const)}>
                          <option value="">Select Party</option>
                          {parties.map(p => <option key={p.id} value={p.id}>{p.name}</option>)}
                        </select>
                        <ErrorMsg error={errors.payments?.[idx]?.partyId} />
                      </div>
                      <div>
                        <select className={selectClass} {...form.register(`payments.${idx}.method` as const)}>
                          <option value="CASH">Cash</option>
                          <option value="CHEQUE">Cheque</option>
                        </select>
                      </div>
                      <div>
                        <input type="number" className={inputClass} placeholder="Amount" {...form.register(`payments.${idx}.amount` as const)} />
                        <ErrorMsg error={errors.payments?.[idx]?.amount} />
                      </div>
                      <div>
                        <input type="text" className={inputClass} placeholder="Note" {...form.register(`payments.${idx}.note` as const)} />
                      </div>
                      {(watchAll.payments[idx]?.method === 'BANK' || watchAll.payments[idx]?.method === 'CHEQUE') && (
                        <div className="sm:col-span-4 mt-2 p-3 bg-[#131b2c] border border-[#1e2d45] rounded-lg">
                          <label className="form-label text-xs flex items-center gap-2">
                            Payslip Attachment (required) 
                            {uploadingAttachment[idx] && <span className="text-[#00d2ff] animate-pulse">Uploading...</span>}
                          </label>
                          {!watchAll.payments[idx]?.attachmentUrl ? (
                            <input 
                              type="file" 
                              accept="image/*,.pdf"
                              disabled={uploadingAttachment[idx]}
                              onChange={async (e) => {
                                const file = e.target.files?.[0]
                                if (!file) return
                                setUploadingAttachment(prev => ({ ...prev, [idx]: true }))
                                try {
                                  const partyId = watchAll.payments[idx]?.partyId || 'unknown'
                                  const timestamp = Date.now()
                                  const filePath = `payslips/${partyId}-${timestamp}-${file.name}`
                                  const { error } = await supabase.storage.from('receipts').upload(filePath, file)
                                  if (error) throw error
                                  const { data } = supabase.storage.from('receipts').getPublicUrl(filePath)
                                  form.setValue(`payments.${idx}.attachmentUrl`, data.publicUrl)
                                } catch (err: any) {
                                  toast.error('Failed to upload attachment: ' + err.message)
                                } finally {
                                  setUploadingAttachment(prev => ({ ...prev, [idx]: false }))
                                }
                              }}
                              className="block w-full text-xs text-[#8899aa] file:mr-4 file:py-1 file:px-3 file:rounded file:border-0 file:text-xs file:font-semibold file:bg-[#10b981] file:text-white hover:file:bg-[#059669] cursor-pointer"
                            />
                          ) : (
                            <div className="flex items-center gap-3">
                              <a href={watchAll.payments[idx].attachmentUrl} target="_blank" rel="noreferrer" className="text-[#00d2ff] hover:underline text-xs flex items-center gap-1">
                                <Paperclip size={14} /> View Attached Payslip
                              </a>
                              <button 
                                type="button"
                                onClick={() => form.setValue(`payments.${idx}.attachmentUrl`, undefined)}
                                className="text-xs text-[#ef4444] hover:underline"
                              >
                                Remove
                              </button>
                            </div>
                          )}
                        </div>
                      )}
                    </div>
                  </div>
                ))}
                <button type="button" onClick={() => appendPayment({ id: generateId(), partyId: '', method: 'CASH', amount: '', note: '', issueDate: '', withdrawDate: '' })} className="text-xs text-[#00d2ff] hover:underline flex items-center gap-1">
                  <Plus size={14} /> Add Payment
                </button>
              </div>

            </div>
          )}
        </div>

        {/* Net Balance & Physical Cash */}
        <div className="bg-gradient-to-br from-[#00d2ff]/10 to-[#3a7bd5]/10 p-5 sm:p-6 rounded-lg border border-[#00d2ff]/30 relative overflow-hidden">
          <div className="absolute top-0 right-0 p-4 opacity-10 pointer-events-none">
            <svg viewBox="0 0 24 24" width="100" height="100" fill="currentColor"><path d="M12 2C6.48 2 2 6.48 2 12s4.48 10 10 10 10-4.48 10-10S17.52 2 12 2zm0 18c-4.41 0-8-3.59-8-8s3.59-8 8-8 8 3.59 8 8-3.59 8-8 8zm.31-8.86c-1.77-.45-2.34-.94-2.34-1.67 0-.84.79-1.43 2.1-1.43 1.38 0 1.9.66 1.94 1.64h1.71c-.05-1.34-.87-2.57-2.49-2.97V5H10.9v1.69c-1.51.32-2.72 1.3-2.72 2.81 0 1.79 1.49 2.69 3.66 3.21 1.95.46 2.34 1.15 2.34 1.87 0 .53-.39 1.64-2.25 1.64-1.74 0-2.4-.91-2.51-1.92H7.72c.12 1.96 1.49 3.33 3.18 3.73V20h2.4v-1.72c1.69-.32 3.01-1.37 3.01-3.08 0-2.39-2.04-3.07-4.04-3.56z"/></svg>
          </div>
          
          <div className="flex flex-col sm:flex-row justify-between items-start sm:items-end gap-6 relative z-10">
            <div>
              <h3 className="text-sm text-[#8899aa] uppercase tracking-widest font-semibold mb-1">System Net Balance</h3>
              <div className="text-4xl font-bold text-white font-mono tracking-tight flex items-baseline gap-2">
                <span className="text-2xl text-[#00d2ff]">৳</span>
                {formatCurrency(totals.netBalance).replace('৳', '')}
              </div>
            </div>

            <div className="w-full sm:w-64">
              <label className="form-label text-xs text-[#00d2ff]">Actual Physical Cash *</label>
              <input 
                type="number" 
                className="form-input w-full h-12 text-lg font-mono focus:ring-2 focus:ring-[#00d2ff] border-[#00d2ff]/50 bg-[#0a0f18]/80"
                placeholder="Enter exact cash amount"
                {...form.register('actualPhysicalCash')}
              />
              <ErrorMsg error={errors.actualPhysicalCash} />
            </div>
          </div>
          
          {Number(watchAll.actualPhysicalCash) > 0 && Math.abs(totals.netBalance - Number(watchAll.actualPhysicalCash)) > 0 && (
            <div className="mt-4 p-4 bg-[#ef4444]/10 border border-[#ef4444]/30 rounded-lg">
              <div className="flex justify-between items-center mb-2">
                <span className="text-sm font-semibold text-[#ef4444]">Discrepancy Detected</span>
                <span className="font-mono font-bold text-[#ef4444]">
                  {formatCurrency(Math.abs(totals.netBalance - Number(watchAll.actualPhysicalCash)))}
                </span>
              </div>
              <input 
                type="text" 
                className="form-input w-full text-sm border-[#ef4444]/30 focus:border-[#ef4444]" 
                placeholder="Reason for cash discrepancy... (Required)"
                {...form.register('cashDifferenceNote')}
              />
            </div>
          )}
        </div>

        {/* Submission Actions */}
        <div className="flex justify-between items-center bg-[#131b2c] p-4 border border-[#1e2d45] rounded-lg mt-8 shadow-2xl sticky bottom-4 z-40">
          <button 
            type="button" 
            className="btn btn-secondary" 
            onClick={() => router.push('/entries')}
            disabled={loading}
          >
            Cancel
          </button>
          <button 
            type="submit" 
            className="btn btn-primary min-w-[200px]"
            disabled={loading || (missingPreviousDay && userRole !== 'ADMIN') || watchAll.payments?.some((p: any) => (p.method === 'BANK' || p.method === 'CHEQUE') && !p.attachmentUrl)}
          >
            {loading ? <span className="spinner" /> : 'Close Register'}
          </button>
        </div>
      </form>

      {/* Checklist Modal */}
      {showChecklistModal && (
        <div className="fixed inset-0 bg-black/60 backdrop-blur-sm flex items-center justify-center p-4 z-50">
          <div className="bg-[#131b2c] border border-[#1e2d45] rounded-xl shadow-2xl max-w-md w-full p-6 animate-in slide-in-from-bottom-4">
            <h3 className="text-xl font-bold mb-4 flex items-center gap-2">
              <Lock className="text-[#00d2ff]" /> Final Checklist
            </h3>
            <p className="text-sm text-[#8899aa] mb-6">
              Please confirm the following closing procedures have been completed before final submission.
            </p>
            
            <div className="space-y-4 mb-8">
              {[
                { id: 'safeLocked', label: 'Safe & registers locked' },
                { id: 'acOff', label: 'A/C and main lights turned off' },
                { id: 'shopClean', label: 'Shop floor cleaned & organized' },
                { id: 'shuttersDown', label: 'Shutters down & padlocked' },
                { id: 'cashVerified', label: 'Physical cash recounted & verified' }
              ].map(item => (
                <label key={item.id} className="flex items-center gap-3 p-3 bg-[#0a0f18] border border-[#1e2d45] rounded-lg cursor-pointer hover:border-[#00d2ff]/50 transition-colors">
                  <input 
                    type="checkbox" 
                    className="w-5 h-5 rounded border-[#1e2d45] bg-[#131b2c] text-[#00d2ff] focus:ring-[#00d2ff] focus:ring-offset-0"
                    {...form.register(`eodChecklist.${item.id}` as any)}
                  />
                  <span className="text-sm select-none">{item.label}</span>
                </label>
              ))}
            </div>

            <div className="flex gap-3">
              <button 
                type="button" 
                className="btn btn-secondary flex-1" 
                onClick={() => setShowChecklistModal(false)}
              >
                Cancel
              </button>
              <button 
                type="button" 
                className="btn btn-primary flex-1" 
                onClick={() => {
                  if (!watchAll.eodChecklist.cashVerified) {
                    toast.error('You must verify physical cash to submit.');
                    return;
                  }
                  setShowChecklistModal(false)
                  handleSubmit(onSubmit)()
                }}
              >
                Close Register
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  )
}
